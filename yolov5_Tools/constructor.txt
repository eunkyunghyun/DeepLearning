from glob import glob

paths = glob("(PATH OF DATASET'S LABEL)", recursive=True)
text = []

for path in paths:
    with open(path, 'r+') as f:
        lines = f.readlines()
        text.clear()
        for line in lines:
            separate = line.split(' ')
            separate[0] = "(SEQUENCE OF DATA)"
            text.append(" ".join(separate))
        f.truncate(0)
        f.seek(0)
        f.writelines(text)
